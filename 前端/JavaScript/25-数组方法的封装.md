# 数组方法的封装（笔记）

## myFindIndex
```js
Array.prototype.myFindIndex = function (func) {
    for (let i = 0; i < this.length; i++) {
        if (func(this[i])) {
            return i;
        }
    }
    return -1
}
let arr = [1, 3, 5, 7]
console.log(arr.myFindIndex((item) => item>= 5)) // 2
```

## myIndexOf
```js
Array.prototype.myIndexOf = function (el) {
    for (let i = 0; i < this.length; i++) {
        if (this[i] === el) {
            return i
        }
    }
    return -1
}
let arr = [1, 3, 5, 7]
console.log(arr.myIndexOf(5)) // 2
```
